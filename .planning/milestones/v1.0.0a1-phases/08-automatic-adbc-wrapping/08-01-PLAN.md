<objective>
Implement the automatic ADBC connection wrapping mechanism. Users configure which driver modules to auto-patch via `adbc_auto_patch` ini key, decorate tests with `@pytest.mark.adbc_cassette`, and call `driver.connect()` normally — the plugin intercepts and returns a `ReplayConnection` transparently.

Purpose: Eliminate the explicit `adbc_replay.wrap()` conftest boilerplate that currently gates plugin adoption.

Output: Updated plugin.py (hooks + fixture), updated _session.py (wrap_from_item), updated _cassette_path.py (driver subdir support).
</objective>

    ```python
    def wrap_from_item(
        self,
        driver_module_name: str,
        item: pytest.Item,
        db_kwargs: dict[str, object] | None = None,
    ) -> ReplayConnection:
        """
        Create a ReplayConnection for a test item (not a FixtureRequest).

        Used by the monkeypatched connect() to resolve cassette path from the
        currently-running test item. Per-driver cassette subdirectory is always
        applied.

        Args:
            driver_module_name: ADBC driver module name (e.g. "adbc_driver_snowflake").
            item: The currently-running pytest test item.
            db_kwargs: Keyword arguments forwarded to the real driver in record mode.

        Returns:
            ReplayConnection ready for use.
        """
        from pytest_adbc_replay._connection import ReplayConnection  # noqa: PLC0415

        resolved_dialect: str | None = self.dialect

        marker = item.get_closest_marker("adbc_cassette")
        if marker is not None:
            resolved_dialect = marker.kwargs.get("dialect", resolved_dialect)
            if marker.args:
                # Named cassette: cassette_dir / name / driver_module_name
                cassette_path = self.cassette_dir / str(marker.args[0]) / driver_module_name
            else:
                # No name: derive from node ID with driver subdir
                cassette_path = node_id_to_cassette_path(
                    item.nodeid, self.cassette_dir, driver_module_name=driver_module_name
                )
        else:
            cassette_path = node_id_to_cassette_path(
                item.nodeid, self.cassette_dir, driver_module_name=driver_module_name
            )

        return ReplayConnection(
            driver_module_name=driver_module_name,
            db_kwargs=db_kwargs or {},
            mode=self.mode,
            cassette_path=cassette_path,
            dialect=resolved_dialect,
            param_serialisers=self.param_serialisers,
        )
    ```

    def pytest_sessionstart(session: pytest.Session) -> None:
        """Monkeypatch ADBC driver connect() for each driver in adbc_auto_patch."""
        import importlib as _importlib

        raw: str = cast("str", session.config.getini("adbc_auto_patch")) or ""
        driver_names = [d.strip() for d in raw.split() if d.strip()]

        for driver_name in driver_names:
            try:
                driver_mod = _importlib.import_module(driver_name)
            except ImportError:
                # Driver not installed — skip silently (may be replay-only environment)
                continue

            original_connect = driver_mod.connect
            _ORIGINAL_CONNECTS[driver_name] = original_connect

            def _make_patched(dn: str, orig: Any) -> Any:
                def _patched_connect(**kwargs: Any) -> Any:
                    with _ITEM_LOCK:
                        item = _CURRENT_ITEM
                    if item is None:
                        return orig(**kwargs)
                    marker = item.get_closest_marker("adbc_cassette")
                    if marker is None:
                        return orig(**kwargs)
                    # Retrieve the session-scoped ReplaySession
                    session_obj: ReplaySession = item.session._adbc_replay_session  # type: ignore[attr-defined]
                    conn = session_obj.wrap_from_item(dn, item, db_kwargs=kwargs)
                    with _ITEM_LOCK:
                        _OPEN_CONNECTIONS.setdefault(id(item), []).append(conn)
                    return conn
                return _patched_connect

            driver_mod.connect = _make_patched(driver_name, original_connect)
    ```

    def pytest_runtest_setup(item: pytest.Item) -> None:
        """Track the currently-running test item for monkeypatched connect() resolution."""
        global _CURRENT_ITEM
        with _ITEM_LOCK:
            _CURRENT_ITEM = item

    def pytest_runtest_teardown(item: pytest.Item, nextitem: pytest.Item | None) -> None:  # noqa: ARG001
        """Clear current item and close all connections opened during this test."""
        global _CURRENT_ITEM
        with _ITEM_LOCK:
            _CURRENT_ITEM = None
        connections = _OPEN_CONNECTIONS.pop(id(item), [])
        for conn in connections:

    @pytest.fixture
    def adbc_connect(
        adbc_replay: ReplaySession,
        request: pytest.FixtureRequest,
    ) -> Any:
        """
        Function-scoped factory fixture for creating ADBC replay connections.

        Escape hatch for session/module-scoped connections or when explicit
        control is needed. Uses per-driver cassette subdirectory layout.

        Usage::

            def test_my_query(adbc_connect):
                conn = adbc_connect("adbc_driver_snowflake", uri=os.environ["SNOWFLAKE_URI"])
                cursor = conn.cursor()
                cursor.execute("SELECT 1")

        Returns:
            A callable: (driver_module_name: str, **db_kwargs) -> ReplayConnection
        """
        opened: list[Any] = []

        def _factory(driver_module_name: str, **db_kwargs: Any) -> Any:
            conn = adbc_replay.wrap_from_item(
                driver_module_name,
                request.node,
                db_kwargs=db_kwargs,
            )
            opened.append(conn)
            return conn

        yield _factory

        for conn in opened:
            try:
                conn.close()
            except Exception:  # noqa: BLE001
                pass
    ```
