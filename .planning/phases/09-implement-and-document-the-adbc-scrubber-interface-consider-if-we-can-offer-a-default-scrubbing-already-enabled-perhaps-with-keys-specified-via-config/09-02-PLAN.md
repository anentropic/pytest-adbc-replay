    class TestScrubKeysIntegration:

    def test_global_scrub_key_redacted_in_cassette(pytester):
        """adbc_scrub_keys = password causes value to become REDACTED in .json cassette."""
        pytester.makepyprojecttoml("""
        [tool.pytest.ini_options]
        adbc_scrub_keys = ["password"]
        """)
        pytester.makepyfile("""
        import pytest
        @pytest.mark.adbc_cassette("scrub_test")
        def test_scrub(adbc_replay, request):
            conn = adbc_replay.wrap(
                "adbc_driver_sqlite.dbapi",
                db_kwargs={"uri": "file::memory:"},
                request=request,
            )
            cur = conn.cursor()
            cur.execute("SELECT ?", {"password": "secret123", "user": "alice"})
            conn.close()
        """)
        result = pytester.runpytest("--adbc-record=once", "-v")
        result.assert_outcomes(passed=1)
        # Verify cassette JSON contains REDACTED
        import json, pathlib
        cassette_files = list(pytester.path.glob("tests/cassettes/scrub_test/*_params.json"))
        assert cassette_files, f"No cassette params found in {list(pytester.path.glob('**/*.json'))}"
        data = json.loads(cassette_files[0].read_text())
        assert data.get("password") == "REDACTED", f"Expected REDACTED, got: {data}"
        assert data.get("user") == "alice", "Non-scrubbed key should be unchanged"

    def test_adbc_scrubber_fixture_called_with_driver_name(pytester):
        """adbc_scrubber callable receives (params, driver_name) at record time."""
        pytester.makepyfile(conftest="""
        import pytest
        CALLS = []
        @pytest.fixture(scope="session")
        def adbc_scrubber():
            def scrub(params, driver_name):
                CALLS.append((params, driver_name))
                return params
            return scrub
        """)
        pytester.makepyfile("""
        import pytest
        @pytest.mark.adbc_cassette("fixture_test")
        def test_scrubber_called(adbc_replay, request):
            conn = adbc_replay.wrap(
                "adbc_driver_sqlite.dbapi",
                db_kwargs={"uri": "file::memory:"},
                request=request,
            )
            cur = conn.cursor()
            cur.execute("SELECT ?", {"x": 1})
            conn.close()
        """)
        result = pytester.runpytest("--adbc-record=once", "-v")
        result.assert_outcomes(passed=1)
        # If scrubber was called, no error raised — success is that test passed
        # (we can't inspect CALLS from outside pytester, but a TypeError would fail the test)

    def test_per_driver_scrub_key_targets_correct_driver(pytester):
        """adbc_scrub_keys per-driver line only redacts for matching driver."""
        pytester.makepyprojecttoml("""
        [tool.pytest.ini_options]
        adbc_scrub_keys = ["adbc_driver_sqlite.dbapi: secret_key"]
        """)
        pytester.makepyfile("""
        import pytest
        @pytest.mark.adbc_cassette("per_driver_test")
        def test_per_driver(adbc_replay, request):
            conn = adbc_replay.wrap(
                "adbc_driver_sqlite.dbapi",
                db_kwargs={"uri": "file::memory:"},
                request=request,
            )
            cur = conn.cursor()
            cur.execute("SELECT ?", {"secret_key": "topsecret", "other": "visible"})
            conn.close()
        """)
        result = pytester.runpytest("--adbc-record=once", "-v")
        result.assert_outcomes(passed=1)
        import json, pathlib
        cassette_files = list(pytester.path.glob("tests/cassettes/per_driver_test/*_params.json"))
        if cassette_files:
            data = json.loads(cassette_files[0].read_text())
            assert data.get("secret_key") == "REDACTED"
            assert data.get("other") == "visible"

    def test_list_params_not_scrubbed(pytester):
        """List/positional params are silently skipped — no key names to match."""
        pytester.makepyprojecttoml("""
        [tool.pytest.ini_options]
        adbc_scrub_keys = ["token"]
        """)
        pytester.makepyfile("""
        import pytest
        @pytest.mark.adbc_cassette("list_params_test")
        def test_list_params(adbc_replay, request):
            conn = adbc_replay.wrap(
                "adbc_driver_sqlite.dbapi",
                db_kwargs={"uri": "file::memory:"},
                request=request,
            )
            cur = conn.cursor()
            # Positional params — no keys to scrub
            cur.execute("SELECT 1")
            conn.close()
        """)
        result = pytester.runpytest("--adbc-record=once", "-v")
        result.assert_outcomes(passed=1)
