```yaml
name: Release

on:
  push:
    tags:
      - 'v[0-9]+.[0-9]+.[0-9]+'
      - 'v[0-9]+.[0-9]+.[0-9]+-*'

permissions:
  contents: read

jobs:
  quality:
    name: Quality gates
    uses: ./.github/workflows/_test.yml

  build:
    name: Build distributions
    needs: [quality]
    runs-on: ubuntu-latest
    timeout-minutes: 10
    steps:
      - name: Checkout repository
        uses: actions/checkout@v4
        with:
          fetch-depth: 0

      - name: Set up uv
        uses: astral-sh/setup-uv@v7
        with:
          enable-cache: true

      - name: Sync dependencies
        run: uv sync --locked --dev

      - name: Build wheel and sdist
        run: uv build

      - name: Upload distributions
        uses: actions/upload-artifact@v4
        with:
          name: dist
          path: dist/
          retention-days: 7

  validate:
    name: Validate distributions
    needs: [build]
    runs-on: ubuntu-latest
    timeout-minutes: 10
    strategy:
      fail-fast: false
      matrix:
        python-version: ["3.11", "3.14"]
    steps:
      - name: Set up uv
        uses: astral-sh/setup-uv@v7
        with:
          python-version: ${{ matrix.python-version }}

      - name: Download distributions
        uses: actions/download-artifact@v4
        with:
          name: dist
          path: dist/

      - name: Test wheel installation
        run: |
          uv venv --python ${{ matrix.python-version }} test-env-wheel
          uv pip install --python test-env-wheel/bin/python dist/*.whl
          test-env-wheel/bin/python -c "import pytest_adbc_replay; print('Wheel installation OK')"

      - name: Test sdist installation
        run: |
          uv venv --python ${{ matrix.python-version }} test-env-sdist
          uv pip install --python test-env-sdist/bin/python dist/*.tar.gz
          test-env-sdist/bin/python -c "import pytest_adbc_replay; print('Sdist installation OK')"

      - name: Check py.typed marker in wheel
        run: |
          unzip -l dist/*.whl | grep "pytest_adbc_replay/py.typed" || (echo "ERROR: py.typed marker missing" && exit 1)

  changelog:
    name: Generate changelog
    needs: [build]
    runs-on: ubuntu-latest
    timeout-minutes: 5
    steps:
      - name: Checkout repository
        uses: actions/checkout@v4
        with:
          fetch-depth: 0

      - name: Install git-cliff
        run: |
          curl -L https://github.com/orhun/git-cliff/releases/download/v2.7.0/git-cliff-2.7.0-x86_64-unknown-linux-gnu.tar.gz | tar xz -C /usr/local/bin

      - name: Generate changelog
        run: git-cliff --latest --output CHANGELOG.md

      - name: Upload changelog
        uses: actions/upload-artifact@v4
        with:
          name: changelog
          path: CHANGELOG.md
          retention-days: 7

  publish:
    name: Publish to PyPI
    needs: [validate, changelog]
    runs-on: ubuntu-latest
    timeout-minutes: 5
    environment:
      name: pypi
      url: https://pypi.org/p/pytest-adbc-replay
    permissions:
      id-token: write
      contents: read
    steps:
      - name: Download distributions
        uses: actions/download-artifact@v4
        with:
          name: dist
          path: dist/

      - name: Publish to PyPI
        uses: pypa/gh-action-pypi-publish@release/v1

  github-release:
    name: Create GitHub Release
    needs: [publish, changelog]
    runs-on: ubuntu-latest
    timeout-minutes: 5
    permissions:
      contents: write
    steps:
      - name: Download distributions
        uses: actions/download-artifact@v4
        with:
          name: dist
          path: dist/

      - name: Download changelog
        uses: actions/download-artifact@v4
        with:
          name: changelog
          path: changelog/

      - name: Create GitHub Release
        uses: softprops/action-gh-release@v2
        with:
          files: dist/*
          body_path: changelog/CHANGELOG.md
          draft: false
          prerelease: false
```

Decision notes:
- `softprops/action-gh-release@v2` is the standard action for GitHub Releases (Claude's discretion per CONTEXT.md)
- `github-release` depends on both `publish` (only create release after PyPI succeeds) and `changelog` (needs the artifact)
- `build` depends on `quality` so the distribution is never built if tests fail
- `validate` and `changelog` both depend on `build` — they run in parallel
- `publish` depends on both `validate` and `changelog` — nothing publishes until both complete
- Checkout actions updated to `@v4` for consistency
